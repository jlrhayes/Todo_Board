import PropTypes from 'prop-types';
import { Link } from 'react-router-dom';

/**
 * A board Button
 * @param {number} color a variable
 * @param {number} text a variable
 * @param {number} onAdd a variable
 * @param {number} link a variable
 * @returns {number} a component
 */
const Button = ({color, text, onAdd, link}) => {
   
    return (
        <Link to = {link}>
            <button onClick = {onAdd} style = {{backgroundColor: color}}>{text}</button>
        </Link>
    )   
}

Button.defaultProps= {
    color : 'steelBlue',
    text : 'Add',
}

Button.propTypes= {
    text : PropTypes.string,
    color : PropTypes.string,
    onClick : PropTypes.func
}

export default Button
